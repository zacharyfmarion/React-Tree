# Basic Tree

## Using a vertex data structure

<Playground>
  {() => {
    const vertexMap = new Map([
      ['O', ['E', 'F', 'N']],
      ['E', ['A', 'D']],
      ['F', []],
      ['N', ['G', 'M']],
      ['A', []],
      ['D', ['B', 'C']],
      ['G', []],
      ['M', ['H', 'I', 'J', 'K', 'L']],
      ['B', []],
      ['C', []],
      ['H', []],
      ['I', []],
      ['J', []],
      ['K', []],
      ['L', []],
    ]);
    return <Tree
      width={120}
      height={50}
      rootId="O"
      vertices={vertexMap}
      showLabels={true}
    />
  }}
</Playground>

## With a custom node component

<Playground>
  {() => {
    const RectangleNode = props => (
      <Node {...props}>
        <rect width={4} height={4} fill="#000" stroke="red" style={{ strokeWidth: 1 }} />

    );
    return <Tree width={120} height={50} rootId={14} showLabels={true}>
